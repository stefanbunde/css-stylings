position: static

  • default positioning for all elements -> the element is not positioned and occurs where it normally would in the document
  • you normally wouldn't specify this unless you need to override a positioning that had been previously set
  • .box1 {
    • position: static;
    • background: red;
  • }
  • .box2 {
    • position: static;
    • background: blue;
  • }
  • .box3 {
    • position: static;
    • background: green;
  • }

position: relative

  • relatively positioned elements behave exactly the same way as statically positioned elements
  • you can use offset properties (top, bottom, left, right) to move the elements to where it would normally occur
  • .box1 {
    • position: relative;
    • background: red;
  • }
  • .box2 {
    • position: relative;
    • left: 100px;
    • background: blue;
  • }
  • .box3 {
    • position: revative;
    • background: green;
  • }

position: absolute

  • the element is removed from the document flow and placed exactly where you tell it to go
  • you can put it anywhere and it won't affect or be affected by any other element in the flow
  • like relative elements, absolutely positioned elements respond to offset properties for positioning
  • like relative elements, absolute elements create a new coordinate system for child elements
  • set parent element to position relative so child elements refer to that parents coordinate system
  • .box1 {
    • position: relative;
    • background: red;
  • }
  • .box2 {
    • position: relative;
    • background: blue;
  • }
  • .box3 {
    • position: absolute;
    • top: 50;
    • left: 50;
    • background: green;
  • }
  • .box1 {
    • position: relative;
    • background: red;
  • }
  • .box2 {
    • position: absolute;
    • bottom: 0;
    • right: 0;
    • background: blue;
  • }
  • .box3 {
    • position: absolute;
    • top: 0;
    • left: 0;
    • background: green;
  • }
  • <div class="box1">
    • <div class="box2"></div>
    • <div class="box3"></div>
  • </div>
